def plus(x):
	summ=0
	for a in x[2:]:
		if a==None:
			summ=summ+0
		elif type(a)==int or type(a)==float:
			summ=summ+a
		else:
			summ=summ+0
	return summ
def cummulative_marks(lst):
	if lst == None:
		return None
	else:
		lst0=[]
		for a in lst:
			for b in a[0:1]:
				rollnum=b[1:3]+'p-'+b[3:]
				for c in a[1:2]:
					lst0.append(tuple((rollnum,c,plus(a))))
		return lst0
